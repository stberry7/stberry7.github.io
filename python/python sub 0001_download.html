import pytube
from tkinter import filedialog, messagebox
from tkinter import *

# 유튜브 동영상 URL
video_url = input("다운로드할 유튜브 동영상 URL을 입력하세요: ")

# pytube 라이브러리를 사용하여 비디오 객체 생성
video = pytube.YouTube(video_url)

# 비디오의 최상의 화질 선택
stream = video.streams.get_highest_resolution()

# 다운로드 폴더 선택 다이얼로그 창 열기
root = Tk()
root.withdraw()
download_path = filedialog.askdirectory(title="다운로드 폴더 선택")

# 다운로드 폴더가 선택되지 않은 경우 프로그램 종료
if not download_path:
messagebox.showinfo("알림", "다운로드 폴더가 선택되지 않았습니다. 프로그램을 종료합니다.")
exit()
else:
messagebox.showinfo("알림", f"선택된 다운로드 폴더: {download_path}")

# 비디오 다운로드
print(f"{video.title}을(를) 다운로드합니다...")
stream.download(download_path)
print("다운로드 완료!")
